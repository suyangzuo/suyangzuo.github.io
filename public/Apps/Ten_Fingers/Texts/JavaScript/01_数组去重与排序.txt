const originalArray = [1, 2, 3, 4]; function removeDuplicatesAndSort(arr) { const uniqueArray = arr.filter((item, index) => arr.indexOf(item) === index); return uniqueArray.sort((a, b) => a - b); } const resultArray = removeDuplicatesAndSort(originalArray); console.log('Original array:', originalArray); console.log('Unique and sorted array:', resultArray); const hasDuplicates = originalArray.length !== resultArray.length; console.log('Array had duplicates:', hasDuplicates);
